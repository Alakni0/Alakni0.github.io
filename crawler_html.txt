<script>
    function pobierzDane() {
        console.log("Klik działa");
        fetch('crawler.php')
            .then(response => response.json())
            .then(data => {
                const tresci = data.tresci;
                document.getElementById('tytul').innerText = tresci.tytul;
                document.getElementById('akapit').innerText = tresci.akapit;

                document.querySelector('button').style.display = 'none';
            })
            .catch(error => {
                console.error("Błąd:", error);
            });
    }
</script>
